# B站视频音频转录与智能总结系统 — DDD 架构设计文档

---

## 一、项目概述

### 1.1 项目目标

构建一个基于 FastAPI 的 Web 应用，用户输入 B 站视频地址后，系统自动完成：

```
用户输入URL → 下载音频 → Whisper语音识别 → LLM加标点 → LLM总结 → 展示结果
```

### 1.2 技术栈

| 层级 | 技术选型 |
|------|---------|
| Web 框架 | FastAPI + Jinja2 |
| 视频下载 | yt-dlp |
| 语音识别 | openai-whisper |
| LLM 调用 | OpenAI API / 本地 Ollama |
| 架构模式 | DDD（领域驱动设计） |
| 异步任务 | BackgroundTasks / Celery（可选） |
| 前端交互 | 原生 HTML + SSE 流式输出 |

```python
class SubmitVideoUseCase:
    """用例：提交视频URL，创建处理任务"""
    def __init__(self, task_repo: ProcessingTaskRepository, event_bus: EventBus):
        self._task_repo = task_repo
        self._event_bus = event_bus
  
    async def execute(self, command: SubmitVideoCommand) -> TaskId:
        # 1. 验证URL
        # 2. 创建 ProcessingTask 聚合
        # 3. 持久化
        # 4. 发布 TaskCreatedEvent
        ...

class ProcessVideoUseCase:
    """用例：执行完整的视频处理流水线"""
    def __init__(self, task_repo, downloader, recognizer, llm_service, audio_storage):
        ...
  
    async def execute(self, task_id: TaskId) -> ProcessingTaskResult:
        # 1. 加载任务
        task = await self._task_repo.find_by_id(task_id)
      
        # 2. 下载音频
        task.start_download()
        audio_bytes, metadata = await self._downloader.download_audio(task.video_url)
        audio_file = await self._audio_storage.save_audio(task_id, audio_bytes, metadata.format)
        task.complete_download(VideoAudio(audio_file, task.video_url))
        await self._task_repo.save(task)
      
        # 3. 语音转录
        task.start_transcription()
        raw_transcript = await self._recognizer.transcribe(audio_file.file_path)
        task.complete_transcription(Transcription(raw_transcript))
        await self._task_repo.save(task)
      
        # 4. LLM加标点
        task.start_text_processing()
        punctuated = await self._llm_service.add_punctuation(raw_transcript.text)
        task.complete_punctuation(punctuated)
        await self._task_repo.save(task)
      
        # 5. LLM总结
        summary = await self._llm_service.summarize(punctuated.text)
        task.complete_summarization(summary)
        await self._task_repo.save(task)
      
        return ProcessingTaskResult.from_task(task)
```

```python
class YtDlpVideoDownloaderAdapter(VideoDownloaderPort):
    """yt-dlp 音频下载适配器（防腐层）"""
  
    async def download_audio(self, url: VideoUrl) -> tuple[bytes, AudioMetadata]:
        output_path = f"/tmp/{url.video_id}"
        ydl_opts = {
            'format': 'bestaudio/best',
            'postprocessors': [{
                'key': 'FFmpegExtractAudio',
                'preferredcodec': 'mp3',
                'preferredquality': '192',
            }],
            'outtmpl': output_path,
            'quiet': True,
        }
      
        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            info = ydl.extract_info(url.value, download=True)
            audio_path = f"{output_path}.mp3"
            audio_bytes = Path(audio_path).read_bytes()
            metadata = AudioMetadata(
                title=info.get('title', ''),
                duration=info.get('duration', 0),
            )
            return audio_bytes, metadata


class WhisperSpeechRecognizerAdapter(SpeechRecognizerPort):
    """Whisper 语音识别适配器（防腐层）"""
  
    def __init__(self, model_size: str = "base"):
        self._model = whisper.load_model(model_size)
  
    async def transcribe(self, audio_path: str, language: str = "zh") -> RawTranscript:
        result = self._model.transcribe(audio_path, language=language)
        segments = [
            TranscriptSegment(start=seg['start'], end=seg['end'], text=seg['text'])
            for seg in result['segments']
        ]
        return RawTranscript(
            text=result['text'],
            language=result.get('language', language),
            segments=segments,
        )


class OpenAILLMAdapter(LLMServicePort):
    """OpenAI 兼容 LLM 适配器（防腐层）"""
  
    def __init__(self, api_key: str, base_url: str = None, model: str = "gpt-4o-mini"):
        self._client = AsyncOpenAI(api_key=api_key, base_url=base_url)
        self._model = model
  
    PUNCTUATION_PROMPT = """请为以下语音识别生成的无标点文本添加合适的标点符号和段落分隔。
要求：
1. 保持原文内容不变，仅添加标点
2. 根据语义适当分段
3. 不要添加任何解释或额外内容

原始文本：
{text}"""
  
    SUMMARY_PROMPT = """请对以下文本进行简洁的总结，提取关键信息。
要求：
1. 总结应涵盖主要观点
2. 使用清晰的结构（可使用序号或小标题）
3. 长度控制在原文的20%-30%

文本：
{text}"""
  
    async def add_punctuation(self, raw_text: str) -> PunctuatedText:
        response = await self._client.chat.completions.create(
            model=self._model,
            messages=[{"role": "user", "content": self.PUNCTUATION_PROMPT.format(text=raw_text)}],
        )
        return PunctuatedText(text=response.choices[0].message.content)
  
    async def summarize(self, text: str) -> TextSummary:
        response = await self._client.chat.completions.create(
            model=self._model,
            messages=[{"role": "user", "content": self.SUMMARY_PROMPT.format(text=text)}],
        )
        content = response.choices[0].message.content
        return TextSummary(text=content, word_count=len(content))
```

```python
    return JSONResponse(status_code=400, content={"error": str(exc)})

@app.exception_handler(TaskNotFoundError)
async def handle_not_found(request, exc):
    return JSONResponse(status_code=404, content={"error": "任务不存在"})

    return JSONResponse(status_code=500, content={"error": str(exc)})
```
